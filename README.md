## Install

```bash
npm install --save react-img-annotation
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'react-img-annotation'
import 'react-img-annotation/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
